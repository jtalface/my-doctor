### 3.2 Core Component Specifications

#### 3.2.1 `<PromptPanel />`

**Purpose:** Displays the current state machine prompt/question to the user.

```typescript
interface PromptPanelProps {
  title: string;                    // Main prompt text
  subtitle?: string;                // Additional context/explanation
  icon?: ReactNode;                 // Optional icon
  variant?: 'default' | 'warning';  // Styling variant
  isLoading?: boolean;              // Show skeleton loading state
}

// Events: None (display-only)

// State Machine Binding:
// - Reads `currentNode.prompt` from orchestrator
// - Reads `currentNode.helpText` for subtitle
```

**Usage:**
```tsx
<PromptPanel 
  title="Have you experienced any chest pain in the past 30 days?"
  subtitle="This helps us understand your cardiovascular health."
/>
```

#### 3.2.2 `<UserInput />`

**Purpose:** Renders the appropriate input component based on node type.

```typescript
interface UserInputProps {
  inputType: 'choice' | 'text' | 'structured' | 'none';
  choices?: string[];
  placeholder?: string;
  validation?: ValidationRule[];
  value?: string | string[];
  onChange: (value: string | string[]) => void;
  onSubmit: () => void;
  isLoading?: boolean;
  disabled?: boolean;
}

interface ValidationRule {
  type: 'required' | 'minLength' | 'maxLength' | 'pattern';
  value?: number | string | RegExp;
  message: string;
}

// Events:
// - onChange: Fires when input value changes
// - onSubmit: Fires when user submits input

// State Machine Binding:
// - Reads `currentNode.inputType` to determine variant
// - Reads `currentNode.choices` for choice buttons
// - Writes user input to orchestrator via handleInput()
```

interface SessionState {
  // Session data

### 7.4 Session Flow Implementation

```typescript
// src/hooks/useSession.ts

export function useSession() {
  const store = useSessionStore();
  
  const handleStartSession = async (type: string) => {
    await store.startSession(userId, type);
    navigate(`/checkup/session/${store.sessionId}`);
  };
  
  const handleInput = async (input: string) => {
    await store.submitInput(input);
    
    // Check for red flags
    if (store.redFlags.length > 0) {
      const critical = store.redFlags.find(f => f.severity === 'high');
      if (critical) {
        navigate('/checkup/alert');
        return;
      }
    }
    
    // Check for terminal state
    if (store.isTerminal) {
      const summary = await store.endSession();
      navigate(`/checkup/summary/${summary.id}`);
    }
  };
  
  const handleBack = () => {
    store.goBack();
  };
  
  return {
    ...store,
    handleStartSession,
    handleInput,
    handleBack,
  };
}
```

export function validateInput(
  input: string,
  nodeType: string,
  rules?: ValidationRule[]
): ValidationResult {
  const errors: string[] = [];
  
  // Type-specific validation
  switch (nodeType) {
    case 'choice':
      if (!input) {
        errors.push('Please select an option to continue');
      }
      break;
      
    case 'text':
      if (!input.trim()) {
        errors.push('Please enter a response');
      }
      if (input.length < 10) {
        errors.push('Please provide more detail (at least 10 characters)');
      }
      if (input.length > 500) {
        errors.push('Response is too long (max 500 characters)');
      }
      break;

  });
  
  return {
    isValid: errors.length === 0,
